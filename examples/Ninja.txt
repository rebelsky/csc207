public class Zebra {
    int value;

    public Zebra() { increment(1); }

    public void increment(int i) { value += i; }
} // class Zebra

public class Stripes extends Zebra {
    public Stripes() { increment(1); }

    public void increment(int i) { value += 3*i; }

    public static int increment(Zebra f) {
        f.increment(12);
        return f.value;
    }

    public static void main(String[] args) {
        System.out.println(increment(new Stripes()));
    } 
} // class Stripes
